query getFreelancerProfile($profileKey: String!) {
  freelancerProfileByProfileKey(profileKey: $profileKey) {
    personalData {
      title
      description
    }
    employmentRecords {
      companyName
      description
      role
      startDate
      endDate
    }
    skills {
      edges {
        node {
          id
        }
      }
    }
    certificates {
      notes
    }
    educationRecords {
      degree
      areaOfStudy
    }
  }
}
